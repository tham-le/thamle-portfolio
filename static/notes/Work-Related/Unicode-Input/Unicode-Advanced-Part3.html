│  │ Layer 2: Client Input Handler                                  │ │
│  │ Location: src/client/input_handler.rs (or platform module)     │ │
│  ├───────────────────────────────────────────────────────────────┤ │
│  │ fn on_character_input(character: char) {                       │ │
│  │     let unicode = UnicodeInput::from_char(character);          │ │
│  │     let payload = Payload::Unicode(unicode);                   │ │
│  │     let packet = InputPacket::new(InputTarget::Host, payload); │ │
│  │     self.send_to_router(packet);                               │ │
│  │ }                                                               │ │
│  └────────────────────────┬────────────────────────────────────────┘ │
│                            │                                          │
│                            │ InputPacket                              │
│                            ▼                                          │
│  ┌───────────────────────────────────────────────────────────────┐ │
│  │ Layer 3: Input Router (Client Side)                            │ │
│  │ Location: src/router/mod.rs                                     │ │
│  ├───────────────────────────────────────────────────────────────┤ │
│  │ fn route_packet(packet: InputPacket) {                         │ │
│  │     match packet.get_target() {                                │ │
│  │         InputTarget::Host => {                                 │ │
│  │             // Apply filters if needed                         │ │
│  │             if packet.filterable() {                           │ │
│  │                 if !self.apply_filters(&packet) {              │ │
│  │                     return; // Blocked                         │ │
│  │                 }                                              │ │
│  │             }                                                  │ │
│  │             self.network.send(packet);                         │ │
│  │         }                                                       │ │
│  │         InputTarget::Client => { /* Local processing */ }      │ │
│  │     }                                                           │ │
│  │ }                                                               │ │
│  └────────────────────────┬────────────────────────────────────────┘ │
│                            │                                          │
│                            │ InputPacket                              │
│                            ▼                                          │
│  ┌───────────────────────────────────────────────────────────────┐ │
│  │ Layer 4: Network Serialization                                 │ │
│  │ Location: src/network/serialization.rs                         │ │
│  ├───────────────────────────────────────────────────────────────┤ │
│  │ fn serialize_for_network(packet: &InputPacket) -> Vec<u8> {   │ │
│  │     let mut buf = Vec::new();                                  │ │
│  │     buf.push(packet.get_type() as u8);  // Type: 10 (Unicode) │ │
│  │     buf.push(packet.get_target() as u8); // Target: 1 (Host)  │ │
│  │     buf.extend(packet.serialize());      // Payload bytes     │ │
│  │     buf                                                        │ │
│  │ }                                                               │ │
│  │                                                                 │ │
│  │ Wire Format:                                                    │ │
│  │ [10][1][0x2D][0x4E][0x00][0x00][0x00]                         │ │
│  │  │   │   └────────┬────────────┘   └──┘                       │ │
│  │  │   │      UTF-32 '中' (0x4E2D)  Modifiers (0x00)            │ │
│  │  │   Target: Host                                              │ │
│  │  Type: Unicode                                                 │ │
│  └────────────────────────┬────────────────────────────────────────┘ │
└─────────────────────────────┼──────────────────────────────────────────┘
                              │
                 ╔════════════▼════════════╗
                 ║   Network Transport     ║
                 ║   TCP/UDP/WebRTC/etc    ║
                 ╚════════════╤════════════╝
                              │
┌─────────────────────────────▼──────────────────────────────────────────┐
│                       HOST COMPUTER (Windows PC)                        │
├─────────────────────────────────────────────────────────────────────────┤
│                                                                          │
│  ┌───────────────────────────────────────────────────────────────────┐ │
│  │ Layer 5: Network Deserialization                                  │ │
│  │ Location: src/network/receiver.rs                                 │ │
│  ├───────────────────────────────────────────────────────────────────┤ │
│  │ fn deserialize_from_network(data: &[u8]) -> Result<InputPacket> {│ │
│  │     let type_ = data[0];        // 10 (Unicode)                   │ │
│  │     let target = match data[1] {                                  │ │
│  │         0 => InputTarget::Client,                                 │ │
│  │         1 => InputTarget::Host,                                   │ │
│  │         _ => return Err(Error::InvalidTarget),                    │ │
│  │     };                                                             │ │
│  │     let payload_data = &data[2..];                                │ │
│  │     InputPacket::deserialize(type_, target, payload_data)         │ │
│  │ }                                                                  │ │
│  └────────────────────────┬─────────────────────────────────────────┘ │
│                            │                                           │
│                            │ InputPacket                               │
│                            ▼                                           │
│  ┌───────────────────────────────────────────────────────────────────┐ │
│  │ Layer 6: Input Router (Host Side)                                 │ │
│  │ Location: src/router/mod.rs                                        │ │
│  ├───────────────────────────────────────────────────────────────────┤ │
│  │ fn route_packet(packet: InputPacket) {                            │ │
│  │     match packet.get_target() {                                   │ │
│  │         InputTarget::Host => {                                    │ │
│  │             // This is for us, send to handler                    │ │
│  │             self.host_handler.handle(packet);                     │ │
│  │         }                                                          │ │
│  │         InputTarget::Client => {                                  │ │
│  │             // Wrong target, ignore or forward back               │ │
│  │         }                                                          │ │
│  │     }                                                              │ │
│  │ }                                                                  │ │
│  └────────────────────────┬─────────────────────────────────────────┘ │
│                            │                                           │
│                            │ InputPacket                               │
│                            ▼                                           │
│  ┌───────────────────────────────────────────────────────────────────┐ │
│  │ Layer 7: Host Input Handler                                       │ │
│  │ Location: src/host/handler.rs                                     │ │
│  ├───────────────────────────────────────────────────────────────────┤ │
│  │ fn handle_packet(packet: InputPacket) {                           │ │
│  │     match packet.payload {                                        │ │
│  │         Payload::Keyboard(key) => {                               │ │
│  │             self.injector.inject_keyboard(&key);                  │ │
│  │         }                                                          │ │
│  │         Payload::Unicode(unicode) => {                            │ │
│  │             self.injector.inject_unicode(&unicode);  // <-- NEW   │ │
│  │         }                                                          │ │
│  │         // ... other payload types                                │ │
│  │     }                                                              │ │
│  │ }                                                                  │ │
│  └────────────────────────┬─────────────────────────────────────────┘ │
│                            │                                           │
│                            │ UnicodeInput                              │
│                            ▼                                           │
│  ┌───────────────────────────────────────────────────────────────────┐ │
│  │ Layer 8: Platform Injector                                        │ │
│  │ Location: src/platform/windows/injector.rs (or linux/macos)      │ │
│  ├───────────────────────────────────────────────────────────────────┤ │
│  │ fn inject_unicode(unicode: &UnicodeInput) -> Result<()> {        │ │
│  │     // Convert UTF-32 to platform encoding                        │ │
│  │     let codepoint = unicode.character as u32;                     │ │
│  │                                                                    │ │
│  │     // Handle modifiers                                           │ │
│  │     if unicode.has_ctrl() {                                       │ │
│  │         press_modifier_key(VK_CONTROL);                           │ │
│  │     }                                                              │ │
│  │                                                                    │ │
│  │     // Platform-specific injection                                │ │
│  │     #[cfg(windows)]                                               │ │
│  │     inject_unicode_windows(codepoint, unicode.modifiers);         │ │
│  │                                                                    │ │
│  │     // Release modifiers                                          │ │
│  │     if unicode.has_ctrl() {                                       │ │
│  │         release_modifier_key(VK_CONTROL);                         │ │
│  │     }                                                              │ │
│  │                                                                    │ │
│  │     Ok(())                                                         │ │
│  │ }                                                                  │ │
│  └────────────────────────┬─────────────────────────────────────────┘ │
│                            │                                           │
│                            │ OS API Call                               │
│                            ▼                                           │
│  ┌───────────────────────────────────────────────────────────────────┐ │
│  │ Layer 9: OS Input Injection (FFI)                                 │ │
│  │ Location: src/platform/windows/ffi.c                              │ │
│  ├───────────────────────────────────────────────────────────────────┤ │
│  │ int inject_unicode_windows(uint32_t codepoint, uint8_t mods) {   │ │
│  │     INPUT input = {0};                                            │ │
│  │     input.type = INPUT_KEYBOARD;                                  │ │
│  │     input.ki.dwFlags = KEYEVENTF_UNICODE;                         │ │
│  │                                                                    │ │
│  │     // Handle surrogate pairs for codepoints > 0xFFFF             │ │
│  │     if (codepoint <= 0xFFFF) {                                    │ │
│  │         input.ki.wScan = (WORD)codepoint;                         │ │
│  │         SendInput(1, &input, sizeof(INPUT));                      │ │
│  │     } else {                                                       │ │
│  │         // Send high surrogate                                    │ │
│  │         input.ki.wScan = high_surrogate(codepoint);               │ │
│  │         SendInput(1, &input, sizeof(INPUT));                      │ │
│  │         // Send low surrogate                                     │ │
│  │         input.ki.wScan = low_surrogate(codepoint);                │ │
│  │         SendInput(1, &input, sizeof(INPUT));                      │ │
│  │     }                                                              │ │
│  │                                                                    │ │
│  │     return 0;                                                      │ │
│  │ }                                                                  │ │
│  └────────────────────────┬─────────────────────────────────────────┘ │
│                            │                                           │
│                            ▼                                           │
│  ┌───────────────────────────────────────────────────────────────────┐ │
│  │ Windows Kernel                                                    │ │
│  │ Character '中' appears in focused application (Notepad, etc.)     │ │
│  └───────────────────────────────────────────────────────────────────┘ │
│                                                                          │
└──────────────────────────────────────────────────────────────────────────┘

  • Update src/lib.rs: Export UnicodeInput publicly

Client Side (Input Capture)

  • iOS: src/client/ios/input_capture.rs + Objective-C bridge
  • Android: src/client/android/input_capture.rs + JNI bridge

Documentation

This complete architecture ensures that a Unicode character typed on a virtual keyboard travels from the client device, through the network, and gets injected into the host operating system correctly, handling all encoding conversions and platform-specific APIs along the way.
